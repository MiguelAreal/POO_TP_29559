86
87 // Validar campos obrigatórios

95

99 erros.Add("O NIF deve conter exatamente 9 dígitos.");

103 erros.Add("Este NIF já está a ser utilizado.");
104 }
105
106 // Validar contacto

109 erros.Add("O contacto deve conter exatamente 9 dígitos.");
110 }
111
112 if (!ValidarContactoComCodigo())

114 erros.Add("Por favor, insira o código do país e o contacto.");
115 }
116

118 if (dtpNasc.Checked && !FieldValidator.ValidateDate(dtpNasc, imgDataNasc))

121 }
122
123 // Validar tipo de utilizador
124 if (!rdbAdmin.Checked && !rdbCliente.Checked)

126 imgTipoUser.ForeColor = Color.Red;
127 erros.Add("Por favor, selecione um tipo de utilizador.");
128 }
129
130 // Exibir mensagens acumuladas, caso existam
131 if (erros.Count > 0)
