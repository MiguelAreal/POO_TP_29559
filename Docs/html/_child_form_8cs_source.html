405 private void txtSearchItem_TextChanged(object sender, EventArgs e)
406 {
407 try
408 {
409 // Obter o filtro digitado pelo usuário no campo de pesquisa
410 string filtro = txtSearchItem.Text;
411
412 // Obter o nome da coluna da célula atualmente selecionada
413 string coluna = dgvItens.CurrentCell?.OwningColumn?.Name;
414
415 // Obter os índices da célula atualmente selecionada (coluna e linha)
416 int colunaIndex = dgvItens.CurrentCell?.ColumnIndex ?? -1; // Verifica se a coluna é válida
417 int rowIndex = dgvItens.CurrentCell?.RowIndex ?? -1; // Verifica se a linha é válida
418
419 // Se o filtro estiver vazio, mostra todos os itens
420 if (string.IsNullOrEmpty(filtro))
421 {
422 MostraItens(_controller.GetItems()); // Método para mostrar todos os itens sem aplicar filtro
423 return;
424 }
425
426 // Filtra os itens com base no filtro e na coluna selecionada
427 List<object> itensFiltrados = _controller.FiltrarItens(_items, filtro, coluna);
428 MostraItens(itensFiltrados);
429
430 // Verifica se há pelo menos uma linha após a filtragem
431 if (itensFiltrados.Count > 0)
432 {
433 // Caso haja itens filtrados, restaura a célula atual
434 if (rowIndex >= 0 && rowIndex < itensFiltrados.Count)
435 {
436 // Se o rowIndex original estiver dentro do intervalo das linhas filtradas
437 dgvItens.CurrentCell = dgvItens.Rows[rowIndex].Cells[colunaIndex];
438 }
439 else
440 {
441 // Se o rowIndex original não existir mais, foca na primeira linha
442 dgvItens.CurrentCell = dgvItens.Rows[0].Cells[colunaIndex];
443 }
444 }
445 else
446 {
447 // Se não houver resultados, desabilita a célula atual
448 dgvItens.CurrentCell = null;
449 }
450 }
451 catch (Exception ex)
452 {
453 // Se ocorrer um erro durante a filtragem, apenas retorna sem fazer nada
454 // Pode ser interessante registrar o erro para depuração
455 // Console.WriteLine($"Erro ao filtrar itens: {ex.Message}");
456 return;
457 }
458 }
