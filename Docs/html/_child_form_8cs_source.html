370 private void txtSearchItem_TextChanged(object sender, EventArgs e)
371 {
372 try
373 {
374 // Obter o filtro digitado pelo usuário no campo de pesquisa
375 string filtro = txtSearchItem.Text;
376
377 // Obter o nome da coluna da célula atualmente selecionada
378 string coluna = dgvItens.CurrentCell?.OwningColumn?.Name;
379
380 // Obter os índices da célula atualmente selecionada (coluna e linha)
381 int colunaIndex = dgvItens.CurrentCell?.ColumnIndex ?? -1; // Verifica se a coluna é válida
382 int rowIndex = dgvItens.CurrentCell?.RowIndex ?? -1; // Verifica se a linha é válida
383
384 // Se o filtro estiver vazio, mostra todos os itens
385 if (string.IsNullOrEmpty(filtro))
386 {
387 MostraItens(_controller.GetItems()); // Método para mostrar todos os itens sem aplicar filtro
388 return;
389 }
390
391 // Filtra os itens com base no filtro e na coluna selecionada
392 List<object> itensFiltrados = _controller.FiltrarItens(_items, filtro, coluna);
393 MostraItens(itensFiltrados);
394
395 // Verifica se há pelo menos uma linha após a filtragem
396 if (itensFiltrados.Count > 0)
397 {
398 // Caso haja itens filtrados, restaura a célula atual
399 if (rowIndex >= 0 && rowIndex < itensFiltrados.Count)
400 {
401 // Se o rowIndex original estiver dentro do intervalo das linhas filtradas
402 dgvItens.CurrentCell = dgvItens.Rows[rowIndex].Cells[colunaIndex];
403 }
404 else
405 {
406 // Se o rowIndex original não existir mais, foca na primeira linha
407 dgvItens.CurrentCell = dgvItens.Rows[0].Cells[colunaIndex];
408 }
409 }
410 else
411 {
412 // Se não houver resultados, desabilita a célula atual
413 dgvItens.CurrentCell = null;
414 }
415 }
416 catch (Exception ex)
417 {
418 // Se ocorrer um erro durante a filtragem, apenas retorna sem fazer nada
419 // Pode ser interessante registrar o erro para depuração
420 // Console.WriteLine($"Erro ao filtrar itens: {ex.Message}");
421 return;
422 }
423 }
