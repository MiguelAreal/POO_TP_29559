117 private void AtualizarTotais()
118 {
119 decimal totalDesconto = 0;
120
121 foreach (DataGridViewRow row in dgvFatura.Rows)
122 {
123 if (row.Cells["Quantidade"].Value != null && row.Cells["PrecoUni"].Value != null && row.Cells["IDProduto"].Value != null)
124 {
125 int quantidade = Convert.ToInt32(row.Cells["Quantidade"].Value);
126 decimal precoUnitario = Convert.ToDecimal(row.Cells["PrecoUni"].Value);
127 int produtoId = Convert.ToInt32(row.Cells["IDProduto"].Value);
128
129 totalBruto += quantidade * precoUnitario;
130
131 // Obter categoria do produto pelo ID
132 Produto produto = (Produto)produtoController.GetById(produtoId);
133
134 if (produto != null)
135 {
136 var desconto = ObterDescontoPorCategoria(produto.CategoriaID);
137 totalDesconto += (quantidade * precoUnitario) * (desconto / 100);
138 }
139 }
140 }
141
142 // Calcular o total líquido
143 totalLiquido = totalBruto - totalDesconto;
144
145 txtTotalBruto.Text = totalBruto.ToString("C");
146 txtTotalLiquido.Text = totalLiquido.ToString("C");
147 }

175 private void VerificarCampanhas()
176 {
177 CampanhaRepo campanhaRepo = new();
178 List<Campanha> campanhas = campanhaRepo.GetAll();
179
180 // Obter categorias dos produtos na fatura
181 var categoriasNaFatura = dgvFatura.Rows
182 .Cast<DataGridViewRow>()
183 .Select(row => row.Cells["IDProduto"].Value)
184 .Where(idProduto => idProduto != null)

188 return produto?.CategoriaID;
189 })
190 .Where(categoriaId => categoriaId != null)
191 .Distinct()
192 .ToList();
193
194 // Filtrar campanhas aplicáveis
195 var campanhasAplicaveis = campanhas
196 .Where(c => categoriasNaFatura.Contains(c.CategoriaId) && campanhaRepo.IsCampanhaValida(c))
197 .ToList();

319 private void btnAddProduto_Click(object sender, EventArgs e)
320 {
321 if (cmbProdutos.SelectedItem is not Produto produtoSelecionado)
322 {
323 MessageBox.Show("Selecione um produto antes de adicionar.", "Erro", MessageBoxButtons.OK, MessageBoxIcon.Warning);
324 return;
325 }
326
327 int quantidadeDesejada = (int)nudQtd.Value;
328
329 // Verificar stock disponível considerando o que já foi adicionado na fatura anteriormente, do mesmo produto
330 int quantidadeTotalFatura = dgvFatura.Rows
331 .Cast<DataGridViewRow>()
332 .Where(row => row.Cells["Produto"].Value?.ToString() == produtoSelecionado.Nome)
333 .Sum(row => Convert.ToInt32(row.Cells["Quantidade"].Value));
334
335 int quantidadeDisponivel = (produtoSelecionado.QuantidadeEmStock - quantidadeTotalFatura);
336
337 // Operador ternário para se a quantidade disponível for negativa, mostrar apenas 0, e não números negativos.
338 quantidadeDisponivel = quantidadeDisponivel < 0 ? 0 : quantidadeDisponivel;
339
340 if (quantidadeDesejada <= 0 || quantidadeDesejada > quantidadeDisponivel)
341 {
342 MessageBox.Show($"Quantidade inválida ou insuficiente no stock! Disponível: {quantidadeDisponivel}", "Erro", MessageBoxButtons.OK, MessageBoxIcon.Warning);
343 return;
344 }
345
346 // Verificar se o produto já está na fatura e atualizar a quantidade na linha
347 // Atualiza totais bruto e liquido.
348 foreach (DataGridViewRow row in dgvFatura.Rows)
349 {
350 if (row.Cells["Produto"].Value?.ToString() == produtoSelecionado.Nome)
351 {
352 int quantidadeAtual = Convert.ToInt32(row.Cells["Quantidade"].Value);
353 row.Cells["Quantidade"].Value = quantidadeAtual + quantidadeDesejada;
354 produtoSelecionado.QuantidadeEmStock -= quantidadeDesejada;
355 return;
356 }
357 }
358
359 // Adicionar novo produto à fatura
360 dgvFatura.Rows.Add(produtoSelecionado.Id, produtoSelecionado.Nome, ObterNomeCategoria(produtoSelecionado.CategoriaID), ObterNomeMarca(produtoSelecionado.MarcaID), quantidadeDesejada, produtoSelecionado.Preco);
361 AtualizarTotais();
362
363 // Verificar campanhas aplicáveis
364 VerificarCampanhas();
365 }
