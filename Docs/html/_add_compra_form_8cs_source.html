140 int quantidadeDesejada = (int)nudQtd.Value;

142 // Verificar stock disponível considerando o que já foi adicionado na fatura anteriormente, do mesmo produto
143 int quantidadeTotalFatura = dgvFatura.Rows
144 .Cast<DataGridViewRow>()
145 .Where(row => row.Cells["Produto"].Value?.ToString() == produtoSelecionado.Nome)
146 .Sum(row => Convert.ToInt32(row.Cells["Quantidade"].Value));
147

149
150 // Operador ternário para se a quantidade disponível for negativa, mostrar apenas 0, e não números negativos.
151 quantidadeDisponivel = quantidadeDisponivel < 0 ? 0 : quantidadeDisponivel;
152
153 if (quantidadeDesejada <= 0 || quantidadeDesejada > quantidadeDisponivel)

155 MessageBox.Show($"Quantidade inválida ou insuficiente no stock! Disponível: {quantidadeDisponivel}", "Erro", MessageBoxButtons.OK, MessageBoxIcon.Warning);

169 }

189 private void AtualizarTotais()
190 {
191 decimal totalDesconto = 0;
192
193 foreach (DataGridViewRow row in dgvFatura.Rows)
194 {
195 if (row.Cells["Quantidade"].Value != null && row.Cells["PrecoUni"].Value != null && row.Cells["IDProduto"].Value != null)
196 {
197 int quantidade = Convert.ToInt32(row.Cells["Quantidade"].Value);
198 decimal precoUnitario = Convert.ToDecimal(row.Cells["PrecoUni"].Value);
199 int produtoId = Convert.ToInt32(row.Cells["IDProduto"].Value);
200
201 totalBruto += quantidade * precoUnitario;
202
203 // Obter categoria do produto pelo ID
204 Produto produto = (Produto)produtoController.GetById(produtoId);
205
206 if (produto != null)
207 {
208 var desconto = ObterDescontoPorCategoria(produto.CategoriaID);
209 totalDesconto += (quantidade * precoUnitario) * (desconto / 100);
210 }
211 }
212 }
213
214 // Calcular o total líquido
215 totalLiquido = totalBruto - totalDesconto;
216
217 txtTotalBruto.Text = totalBruto.ToString("C");
218 txtTotalLiquido.Text = totalLiquido.ToString("C");
219 }

250 private void VerificarCampanhas()
251 {
252 CampanhaRepo campanhaRepo = new();
253 List<Campanha> campanhas = campanhaRepo.GetAll();
254
255 // Obter categorias dos produtos na fatura
256 var categoriasNaFatura = dgvFatura.Rows
257 .Cast<DataGridViewRow>()
258 .Select(row => row.Cells["IDProduto"].Value)
259 .Where(idProduto => idProduto != null)
260 .Select(idProduto =>
261 {
262 Produto produto = (Produto)produtoController.GetById(Convert.ToInt32(idProduto));
263 return produto?.CategoriaID;
264 })
265 .Where(categoriaId => categoriaId != null)
266 .Distinct()
267 .ToList();
268
269 // Filtrar campanhas aplicáveis
270 var campanhasAplicaveis = campanhas
271 .Where(c => categoriasNaFatura.Contains(c.CategoriaId) && campanhaRepo.IsCampanhaValida(c))
272 .ToList();
273
274 AtualizaListViewCampanhas(campanhasAplicaveis);

394 private VendaCompra CriarCompra()

396 VendaCompra venda = new()

398 ClienteID = _cliente.Id,

400 TotalBruto = totalBruto,
401 TotalLiquido = totalLiquido,
402 DataVenda = DateTime.Now.ToString("dd-MM-yyyy - HH:mm"),

404
405 MetodoPagamento = Enum.TryParse(cmbMetodoPagamento.SelectedValue?.ToString(), out MetodoPagamento metodo)
406 ? metodo
407 : (MetodoPagamento?)null
408 };
409
410 foreach (DataGridViewRow row in dgvFatura.Rows)
411 {
412 if (row.Cells["IDProduto"].Value != null && row.Cells["Quantidade"].Value != null)
413 {
414 int produtoID = Convert.ToInt32(row.Cells["IDProduto"].Value);
415 int quantidade = Convert.ToInt32(row.Cells["Quantidade"].Value);
416
417 Produto produto = (Produto)produtoController.GetById(produtoID);
418

424 produto.Preco,
425 produto.CategoriaID,
426 ObterNomeCategoria(produto.CategoriaID),
427 ObterNomeMarca(produto.MarcaID),
428 quantidade,
429 (int?)ObterDescontoPorCategoria(produto.CategoriaID) // Adicionado o argumento para percentagemDesc
430
431 );

436 }
